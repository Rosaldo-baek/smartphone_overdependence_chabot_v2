# =============================================================
# requirements.txt - 스마트폰 과의존 실태조사 RAG 챗봇
# 
# [전략] 코드가 langchain 0.2.x API 기반이므로,
#   langchain 0.2.x 호환 범위 내에서 최대한 안정적인 버전으로 구성.
#   openai SDK는 최신 모델(gpt-5-mini, gpt-5.4-mini 등) 호출이
#   가능하도록 충분히 높은 버전을 사용.
#
# [주의사항]
#   - langchain 1.x로 올리면 import 경로, API 시그니처 등 
#     Breaking Change가 발생하므로 코드 전면 수정 필요
#   - langchain-openai 0.1.25는 langchain-core 0.2.x와 호환되는 
#     0.1.x 계열의 마지막 안정 버전
#   - openai>=1.50.0으로 올려야 최신 모델명 + 
#     text-embedding-3-large 안정 호출 보장
# =============================================================

# ----- Streamlit -----
streamlit==1.32.2

# ----- LangChain 0.2.x 호환 세트 -----
# langchain 0.2.16은 langchain-core >=0.2.38,<0.3.0 을 요구
langchain==0.2.16
langchain-core==0.2.38
langchain-text-splitters==0.2.4

# langchain-openai 0.1.25: 0.1.x 계열 최종 버전
# langchain-core >=0.2,<0.3 호환, openai >=1.40 이상 지원
langchain-openai==0.1.25

# langchain-chroma: langchain 0.2.x와 호환
langchain-chroma==0.1.4

# ----- LangGraph -----
# 0.2.x 계열 유지 (코드의 StateGraph, MemorySaver 등 API 호환)
langgraph==0.2.14

# ----- Vector Store -----
chromadb==0.5.3

# ----- OpenAI SDK -----
# 1.50.0 이상: gpt-5-mini, text-embedding-3-large 등 
# 최신 모델 안정 지원 + httpx 호환성 개선
openai>=1.50.0,<2.0.0

# ----- HTTP 클라이언트 -----
# openai SDK 내부에서 사용. 버전 고정 해제하여 openai와 자동 호환
httpx>=0.27.0,<1.0.0

# ----- Hugging Face (Chroma DB 다운로드용) -----
huggingface_hub>=0.24.0,<1.0.0

# ----- 데이터 처리 -----
pandas==2.2.2

# ----- 기타 유틸리티 -----
python-dotenv==1.0.1

# ----- 충돌 방지 -----
# protobuf 3.x: chromadb 내부 grpc 의존성과 호환
protobuf>=3.20.0,<4.0.0

# posthog: langchain 내부 텔레메트리 의존성
posthog>=3.0.0
